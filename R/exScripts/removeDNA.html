files <- list.files("../2_alignedData/bam/", pattern = "bam$", full.names = TRUE)

This data set contains 9 files.

win <- getWinFromBamFile(files, sequences = "9")

For data inspection reads from chromosome 9 were checked for strandedness using a sliding window across all files. A small shoulder of alignments which didnt match the pattern of strict strandedness was detected for most files.

plotHist(
  windows = win, group_by = c("File","OverlapTranscript"), 
  normalize_by = "File", scales = "free_y"
) +
  facet_wrap(~File, labeller = lb)

Filter files

p <- 0.8

The alignments were then filtered to remove any putative gNDA, using a threshold of 0.8.

for (i in seq_along(files)){
  inFile <- files[i]
  outFile <- file.path("../4_filteredData/bam/",
                       basename(inFile))
  statfile <- file.path("../4_filteredData/log/",
                        inFile %>%
                          basename() %>%
                          str_remove("Aligned.+") %>%
                          paste0("filterDNA.log"))
  filterDNA(inFile, outFile, statfile, threshold = p)
}
postBam <- list.files(
  "../4_filteredData/bam/",
  pattern = "bam$",
  full.names = TRUE) %>%
  BamFileList()
postWin <- getWinFromBamFile(postBam, sequences = "9")
plotHist(
  windows = postWin, group_by = c("File","OverlapTranscript"),

) +
